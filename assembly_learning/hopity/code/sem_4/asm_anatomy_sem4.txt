Управление потоком выполнения

Флаги условий

int t = a+b;

CF - Carry Flag = (unsigned) t < (unsigned) a
ZF - Zero Flag = (t == 0)
SF - Sign Flag = (t < 0)
OF - Overflow Flag = (a < 0 == b < 0) && (t < 0 != a < 0)

D - destination

Для знаковых
__________________________________________________________________________________
sete    D    D <- ZF    (equal)        (sete === setz)                            |
setz    D    D <- ZF    (zero)                                                    |
__________________________________________________________________________________|
setne   D    D <- !ZF   (not equal)    (setne === setnz)                          |
setnz   D    D <- !ZF   (not zero)                                                |
__________________________________________________________________________________|
sets    D    D <- SF   (sign, negative)                                           |
setns   D    D <- !SF   (not sign, non-negative)                                  |
__________________________________________________________________________________|
setg    D    D <- ~(SF^OF) & ~ZF    (greater)                (setg === setnle)    |
setnle  D    D <- ~(SF^OF) & ~ZF    (not (less or equal))                         |
__________________________________________________________________________________|
setge   D    D <- ~(SF^OF) (greater or equal)    (setge == setnl)                 |
setnl   D    D <- ~(SF^OF) (not less)                                             |
__________________________________________________________________________________|
setl    D    D <- SF^OF                                                           |
setnge  D    D <- SF^OF                                                           |
__________________________________________________________________________________|
setle   D    D <- (SF^OF)|ZF                                                      |
setng   D    D <- (SF^OF)|ZF                                                      |
__________________________________________________________________________________|

Для беззнаковых

seta    D    D <- ~CF & ~ZF    (above)
setnbe  D    D <- ~CF & ~ZF    (not below or equal)

setae   D    D <- ~CF
setnb   D    D <- ~CF

setb    D    D <- CF
setnge  D    D <- CF

setb    D    D <- CF | ZF
setna   D    D <- CF | ZF
